function printout(arr) {
  //rows表示行数，columns表示列数
  const rows = arr.length;
  if (rows === 0) return [];
  if (rows === 1) return arr[0];
  const columns = arr[0].length;
  if (columns === 1) return arr.map((line) => line[0]);
  
  //定义res来存放新数组
  let res = [];
  while (arr.length > 0) {
    const firstLine = arr.shift();
    res.push(...firstLine);
    if (arr.length && arr[0].length) {
      arr.forEach((line) => {
        res.push(line.pop());
      });
      const lastColumn = arr.pop();
      res.push(...lastColumn.reverse());
      const lastLine = [];
      arr.forEach((line) => {
        lastLine.unshift(line.shift());
      });
      res.push(...lastLine);
    }
  }
  console.log(res);
  return res;
}
printout([
  [1, 2, 3, 4],

]);
