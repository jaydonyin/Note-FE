给定一个 正整数 num ，编写一个函数，如果 num 是一个完全平方数，则返回 true ，否则返回 false 。
进阶：不要 使用任何内置的库函数，如  sqrt 。

```
/**
 * @param {number} num
 * @return {boolean}
 */
var isPerfectSquare = function(num) {
    let start = 0, n = num, resolve = false;
    while(start <= n) {
        const mid = (start + n) >>> 1;
        if(mid**2 === num) {
            resolve = true;
            break;
        }
        const isSmall = (mid**2) < num;
        start = isSmall ? mid+1 : start;
        n = isSmall ? n : mid-1;
    }
    return resolve;
};

```
